/* global define, console, app, $, breeze */
define([
    'underscore',
    'backbone',
], function (_, Backbone) {
    "use strict";

    var Location = Backbone.Model.extend( /** @lends Location.prototype */ {
        /**
         * @class Location Holds location information for navigation and query purposes. Subscribes to {@link User User} to query
         * location data when user logs in (user:login). Accessible via app.location.
         * @example
         * // Get the currently active location
         * app.location.get('active');
         *
         * @augments Backbone.Model
         * @constructs
         * @requires Backbone
         * @param {object} options - Backbone.Model options
         * @property {Collection} locations - Collection holding Location Entity Data
         * @property {number} home - From {@link User User} claims, this is the Id for uppermost location the user can traverse
         * @property {number} homeType - The type of location for home
         * @property {number} active - The id of the currently active location. Default is null (null before user login)
         * @property {number} activeType - The type of location for active
         * @property {Collection} crumbs - A collection with ordered information about the current path from active to home
         * @property {string} crumbs.display - The displayname or "..." for the crumb
         * @property {number} crumbs.parentId - The parentId for the crumb
         * @property {number} crumbs.activeId - The id for the crumb
         * @property {number} crumbs.activeTypeId - The type for the crumb
         * @property {Backbone.Collection} locations - Location data associated with breadcrumbs (holds breeze location entities)
         */
        initialize: function (attributes, options) {
            _.bindAll(this, '_buildCrumbs', 'setActive');

            this.set('locations', app.api.getChildCollection('Location'));

            this.listenTo(Backbone.EventBroker, 'user:login', function (user) {
                this.set('home', user.get('topLevelLocationId'));
                this.set('homeType', user.get('topLevelLocationTypeId'));
                //testing startlocation vs. topLevel location
                //this.setActive(user.get('startLocationId'), user.get('startLocationTypeId'));
                this.setActive(user.get('topLevelLocationId'), user.get('topLevelLocationTypeId'), false);
            });
        },

        title: 'Location Module',

        // TODO: Update locations to use a query engine collection inherting from breeze entity manager.
        defaults: {
            locations: null,
            home: null,
            homeType: null,
            active: null,
            activeType: null,
            crumbs: new Backbone.Collection()
        },	

        /**
         * Internal function for maintaining/updating the crumbs collection on location change (when this.get('active') changes).
         * This function loops internally traversing parents of the activeId param until 'home' is discovered.
         * @private
         * @param {number} [activeId] - the id of the lowermost crumb to be created. (generally will be activeId)
         * @param {number} [activeTypeId] - the id type of the lowermost crumb to be created. (generally will be activeTypeId)
         */
		
        _buildCrumbs: function (activeId, activeTypeId) {
            // We assume we always want to build from current active location (bottom of the crumb chain) unless arguments specify otherwise
            activeId = (activeId) ? parseInt(activeId) : this.get('active');
            activeTypeId = (activeTypeId) ? activeTypeId : this.get('activeType');
            var parentId = activeId;
            var locations = this.get('locations');
            var newCrumbs = [];
            var displayName = "...";
            var buildCrumbs = true;
            while (buildCrumbs) {

                // If we have a parentId, add it to the beginning of the crumb trail.
                if (locations.where({
                    ParentId: parentId
                }).length > 0) newCrumbs.unshift({
                    display: (parentId == activeId) ? "..." : displayName,
                    parentId: parentId,
                    activeId: activeId,
                    activeTypeId: activeTypeId
                });

                // Before continuing the loop, check there is another location up the chain
                // If no location or the location is home, break our loop.
                var location = locations.findWhere({
                    Id: parentId
                });
                if (!location) buildCrumbs = false;
                if (parentId == this.get('home')) buildCrumbs = false;
                else if (location) {
                    // A location was found, update the parameters consumed by our while loop and continue
                    parentId = location.get('ParentId');
                    displayName = location.get('Name');
                    activeId = location.get('Id');
                    activeTypeId = location.get('LocationTypeId');
                }
            }
			// ordering locations by name - this also works when adding new locations
			locations.comparator = 'Name';
			locations.sort();
            // We want to reset our crumbs collection with the array chain we just built.
            this.get('crumbs').reset(newCrumbs);
        },

        /**
         * Sets the currently active location, and queries for children if applicable.  Will call on this._buildCrumbs to update crumbs.
         * TODO: make sure this handles logic related to pathing up a nested chain (for example in future user starting preference option)
         * @param {number} activeId - the id of the location to be set as active
         * @param {number} activeTypeId - the type of the location to be set as active
         * @param {boolean} reloadChildLocations - If true will force reloading of child locations
         */
        setActive: function (activeId, activeTypeId, reloadChildLocations) {
            if (activeId == this.get('active') && !reloadChildLocations) return true;
            app.api.manager.rejectChanges();
            // TODO: If the active item's parent is not in locations (ex: implementation of user starting location preference),
            // we will need to add special call here to properly build our crumbs.  For now, we simply go get children.

            // Manually call creation of the breadcrumbs - we do not update this.active until our breeze callback, 
            // so we manually pass it here.  This allows the crumbs to build immediately (and then update on callback)
            this._buildCrumbs(activeId, activeTypeId);

            var _this = this;
            var locations = this.get('locations');

            // Currently we allow deactivated sites to be displayed in breadcrumbs, otherwise uncomment this:
            // var p1 = breeze.Predicate.create("IsActive",breeze.FilterQueryOp.Equals, true);

            // Build our breeze predicate
            var p2 = app.api.Predicate.create("IsDeleted", app.api.FilterQueryOp.Equals, false);
            var p3 = app.api.Predicate.create("ParentId", app.api.FilterQueryOp.Equals, activeId).or("Id", app.api.FilterQueryOp.Equals, activeId);
            var p4 = app.api.Predicate.create("UserId", app.api.FilterQueryOp.Equals, "'" + app.userId + "'");
            var p5 = app.api.Predicate.create("UserLocations", app.api.FilterQueryOp.Any, p4);
            var wherePredicate = app.api.Predicate.and(p2, p3, p5);

            // Create a standard breeze query
            var query = app.api.breeze
                .EntityQuery
                .from('Locations')
                .expand('Parameters')
				.expand('Address')
				//.orderBy('Name')
            // Utilizing the select returns a custom entity type.  If performance concerns for a full entity become an issue,
            // re-implement the select
            // .select('Name,Id,ParentId,LocationTypeId,IsActive')
            .where(wherePredicate)
                .using(app.api.manager);

            // Prep our callback for our breeze query
            var callback = function (data) {
                _this.set('active', activeId);
                _this.set('activeType', activeTypeId);
                _this._buildCrumbs(activeId, activeTypeId);
                /**
                 * On successful query, trigger change of active location
                 * @event Location#location:change
                 */
                Backbone.EventBroker.trigger('location:change', {
                    LocationId: activeId,
                    ActiveTypeId: activeTypeId
                });
            };

            app.api.get(query, {
                callback: callback
            });
			
        }
    });

    return Location;
});
