    define([
  'underscore',
  'base/view',
  'text!modules/grid/templates/grid_row.html'
], function (_, View, template) {

    "use strict";

    var Row = View.extend({
        template: _.template(template),

        tagName: "tr",

        events : {
            'click td' : 'editMode',
            'blur input' : 'saveChanges'
        },

        editMode: function(event){
            if(!this._editing) $(event.currentTarget).find('span').hide().siblings().fadeIn().focus();
            this._editing = true;
        },

        saveChanges: function(event){
            var $el = $(event.currentTarget);
            $el.hide().siblings().fadeIn();
            this._editing = false;
            app.api.manager.saveChanges();
        },

        createViewModel: function (options) {
            var viewModel = this._createViewModel({
                exclusions: []
            });
            viewModel.columns = this.grid.getViewModel().get('columns');
            return viewModel;
        }
    });

    return Row;
});
